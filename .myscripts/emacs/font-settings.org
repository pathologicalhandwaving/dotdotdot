;; font-settings

;; all-the-icons
(use-package 'all-the-icons)


(all-the-icons-octicon "file-binary") 
;; GitHub Octicon for Binary File

(all-the-icons-faicon  "cogs")
;; FontAwesome icon for cogs

(all-the-icons-wicon   "tornado")
;; Weather Icon for tornado


;; Color example
(all-the-icons-wicon "tornado" :face 'all-the-icons-blue)


;; Properties
(propertize (all-the-icons-octicon "package")
            'face `(:family ,(all-the-icons-octicon-family) :height 1.2)
            'display '(raise -0.1))


;; Helper Functions
;; Print all icons in a font

(all-the-icons-insert-icons-for 'alltheicon) 
;; Prints all the icons for `alltheicon' font set

(all-the-icons-insert-icons-for 'octicon 10) 
;; Prints all the icons for the `octicon' family
;; and makes the icons height 10

(all-the-icons-insert-icons-for 'faicon 1 0.5) 
;; Prints all the icons for the `faicon' family 
;; and also waits 0.5s between printing each one
